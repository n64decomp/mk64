16 CourseVtx *vertexStart; // 0x18 segmented address of vertex data
17 u32 vertexCount; // 0x1C number of vertices in vertex data

19 uintptr_t finalDisplaylistOffset; // 0x24
20 u32 *textures; // 0x20 segmented address of textures table

23};
